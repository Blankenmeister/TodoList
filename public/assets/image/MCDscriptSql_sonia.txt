CREATE TABLE ToDoUser(
   id_User COUNTER,
   nom VARCHAR(50) NOT NULL,
   prenom VARCHAR(50) NOT NULL,
   email VARCHAR(80) NOT NULL,
   MotDePasse VARCHAR(250) NOT NULL,
   PRIMARY KEY(id_User)
);

CREATE TABLE toDoPriority(
   id_Prio COUNTER,
   nom VARCHAR(50) NOT NULL,
   PRIMARY KEY(id_Prio)
);

CREATE TABLE toDoCategory(
   id_Cat COUNTER,
   nom VARCHAR(50),
   PRIMARY KEY(id_Cat)
);

CREATE TABLE toDoTask(
   id_Task COUNTER,
   titre VARCHAR(50) NOT NULL,
   description VARCHAR(50),
   _date DATE NOT NULL,
   id_Prio INT NOT NULL,
   id_User INT NOT NULL,
   PRIMARY KEY(id_Task),
   FOREIGN KEY(id_Prio) REFERENCES toDoPriority(id_Prio),
   FOREIGN KEY(id_User) REFERENCES ToDoUser(id_User)
);

CREATE TABLE toDoCatIntermediaire(
   id_Task INT,
   id_Cat INT,
   id_CatInt VARCHAR(50),
   PRIMARY KEY(id_Task, id_Cat),
   FOREIGN KEY(id_Task) REFERENCES toDoTask(id_Task),
   FOREIGN KEY(id_Cat) REFERENCES toDoCategory(id_Cat)
);
